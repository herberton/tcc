@startuml

	left to right direction

	actor Cliente				as CLI
	actor :Cliente Especial:	as CLI_ESP
	actor :Administrador:		as ADM
	
	rectangle "Cadastramento" {
	
		package ActiveDirectory {
			usecase (Cadastra Usuario)	as CAD_USU
		}
		
		package Portal-WEB  {
			usecase (Cadastra Vendedor)												as CAD_VEN
			usecase (Cadastra Fabricante)											as CAD_FAB
			usecase (Cadastra Revendedor)											as CAD_REV
			usecase (Cadastra Fornecedor)											as CAD_FOR
			usecase (Cadastra Parceiro de Entrega)									as CAD_PAR_ENT	
			usecase (Cadastra Cliente Especial)										as CAD_CLI_ESP
			usecase (Exibe Solicitacoes de Cadastro\nde Cliente Especial Pendentes)	as EXI_SOL_CAD_CLI_ESP_PEN
		}
		
		package E-commerce {
			usecase (Cadastra Cliente)											as CAD_CLI
			usecase (Inclui Solicitacao de Cadastramento\nde Cliente Especial)	as INC_SOL_CAD_CLI_ESP
		}
	
	}
	
	INC_SOL_CAD_CLI_ESP	.> CAD_CLI :<<include>>
	
	CAD_VEN		..> CAD_USU :<<include>>
	CAD_FAB		..> CAD_USU :<<include>>
	CAD_REV		..> CAD_USU :<<include>>
	CAD_FOR		..> CAD_USU :<<include>>
	CAD_PAR_ENT	..> CAD_USU :<<include>>
	CAD_CLI		..> CAD_USU :<<include>>

	CLI --> CAD_CLI
	ADM --> CAD_CLI_ESP
	ADM --> CAD_VEN
	ADM --> CAD_FAB
	ADM --> CAD_REV
	ADM --> CAD_FOR
	ADM --> CAD_PAR_ENT
	
	CLI_ESP -|> CLI
	CLI_ESP --> INC_SOL_CAD_CLI_ESP
	
	CAD_CLI_ESP ..> EXI_SOL_CAD_CLI_ESP_PEN :<<include>>
	EXI_SOL_CAD_CLI_ESP_PEN .> INC_SOL_CAD_CLI_ESP :<<extend>>
	
@enduml
