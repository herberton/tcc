@startuml

	left to right direction

	actor Administrador as ADM
	actor Fabricante	as FAB
	actor Revendedor	as REV
	actor Auditor		as AUD
	
	rectangle "Descarte e Recolhimento" {
	
		package ActiveDirectory {
			usecase (Cadastra Usuario)	as CAD_USU
		}
		
		package Cadastro  {
			usecase (Cadastra Fabricante)									as CAD_FAB
			usecase (Inclui Solicitacao de\nCadastramento de Fabricante)	as INC_SOL_CAD_FAB
			usecase (Lista Solicitacoes de\nCadastramento de Fabricante)	as LIS_SOL_CAD_FAB
			usecase (Cadastra Revendedor)									as CAD_REV
			usecase (Inclui Solicitacao de\nCadastramento de Revendedor)	as INC_SOL_CAD_REV
			usecase (Lista Solicitacoes de\nCadastramento de Revendedor)	as LIS_SOL_CAD_REV
		}
		
		package "API-Gateway EDI" {
			usecase (Recebe Notificacao de\nDescarte de Embalagens)								as REC_NOT_DES_EMB
			usecase (Realiza Descarte de Embalagens)											as REA_DES_EMB
			usecase (Recebe Notificacao para Recolhimento\nde Produtos Agrotoxicos Vencidos)	as REC_NOT_REC_PRO_AGRO_VEN
			usecase (Recolhe Produtos Agrotoxicos Vencidos)										as REC_PRO_AGRO_VEN
		}
		
		package "SCM" {
			usecase (Envia Notificacao de\nDescarte de Embalagens)							as ENV_NOT_DES_EMB
			usecase (Envia Notificacao para Recolhimento\nde Produtos Agrotoxicos Vencidos)	as ENV_NOT_REC_PRO_AGRO_VEN
		}
	
	}
	
	CAD_FAB --|> CAD_USU
	CAD_REV --|> CAD_USU
	
	CAD_FAB ..> LIS_SOL_CAD_FAB :<<include>>
	CAD_REV ..> LIS_SOL_CAD_REV :<<include>>
	
	LIS_SOL_CAD_FAB ..> INC_SOL_CAD_FAB :<<extend>>
	LIS_SOL_CAD_REV ..> INC_SOL_CAD_REV :<<extend>>	
	
	ADM --> CAD_FAB
	ADM --> CAD_REV
	
	FAB -up-> INC_SOL_CAD_FAB
	FAB -up-> REA_DES_EMB
	FAB -up-> REC_PRO_AGRO_VEN
	FAB -up-> REC_NOT_DES_EMB
	
	REV -up-> INC_SOL_CAD_REV
	REV -up-> REC_PRO_AGRO_VEN
	REV <-up- REC_PRO_AGRO_VEN
	REV <-up- REC_NOT_DES_EMB
	
	AUD --> ENV_NOT_DES_EMB
	AUD --> ENV_NOT_REC_PRO_AGRO_VEN
	
	
@enduml
